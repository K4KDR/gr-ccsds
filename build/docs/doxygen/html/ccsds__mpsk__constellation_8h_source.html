21 /*! \brief Converts gray coded number into corresponding binary number.
22  *
23  * \param num Gray coded number to transform into a binary number.
24  * \return Binary coded number
25  *
26  * Code from http://en.wikipedia.org/wiki/Grey_code#Converting_to_and_from_Gray_code
27  */
28 inline unsigned int tobinary(unsigned int num) {
29  unsigned int numBits = 8 * sizeof(num);
30  unsigned int shift;
31  for (shift = 1; shift < numBits; shift = 2 * shift)
32  {
33  num = num ^ (num >> shift);
34  }
35  return num;
36 }

38 /*! \brief Create general M-PSK constellation.
39  *
40  * \param M Modulation order.
41  * \return Pointer to array of \c M constellation elements,
42  * allocated with new. This array must be deallocated with delete[] after
43  * usage.
44  *
45  * Gray coded M-PSK constellation starting with zero symbol on the real axis
46  * and rotating counterclockwise afterwards.
47  */
48 inline gr_complex* ccsds_make_psk_const(unsigned int M) {
49  // create phasor for rotation between the symbols
50  const gr_complex diff_phasor = gr_complex(cos(2*M_PI/M),sin(2*M_PI/M));
51 
52  // initial phasor is on the real axis
53  gr_complex phasor = gr_complex(1.0f,0.0f);
54  // make sure we only have abs(.)=1 phasors
55  phasor = phasor / std::abs(phasor);
56 
57  // allocate memory for the constellation points
58  gr_complex *map = new gr_complex[M];
59 
60  // go through all constellation points
61  for(unsigned int i=0;i<M;i++) {
62  unsigned int i_gray = togray(i);
63 
64  // check if the next access is valid (not true for e.g M=3)
65  if(i_gray >= M) {
66  fprintf(stderr,"ERROR: modulation order M=%d not supported. M needs to be a power of two.\n",M);
67  exit(EXIT_FAILURE);
68  return NULL;
69  }
70  // store constellation point
71  map[i_gray] = phasor;
72  // rotate to next point
73  phasor *= diff_phasor;
74  }
75 
76  return map;
77 }

80 /*! \brief Create QPSK constellation according to ECSS-E-ST-50-05C.
