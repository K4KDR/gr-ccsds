8 /*! \brief Create a linear second order low pass filter and return it's shared
9  * pointer.
10  *
11  * \param loop_bw Bandwidth of the loop. Used to compute the loop stepsizes.
12  * \param damping_factor_squared Squared value of the loop's damping factor.
13  * Used to compute the loop stepsizes.
14  * \param s_slope Estimated slope of the S-curve of the error detector at the
15  * origin. Used to compute the loop stepsizes.
16  * \return Shared pointer of the created filter.
17  *
18  * The loop stepsizes gamma and rho are given by the following formulas:
19  * [\ref bib_mengali p. 220 ff.]
20  * \code
21  * loop_bw = ( 2*rho + gamma*s_slope*(2+rho) ) / ( 8 - 2*gamma*s_slope*(2+rho) ) (5.3.92)
22  * damping_factor = ( (1+rho)*sqrt(gamma*s_slope) ) / ( 2*sqrt(rho)) (5.3.86)
23  * \endcode
24  *
25  * The stepsize rho can not be computed in explicit form and is computed with
26  * Newton's method by solving the root of the following polynomial:
27  * \code f(rho) = rho^3 + a * rho^2 + b * rho + c != 0 \endcode
28  * with
29  * \code
30  * a = - 4*loop_bw + 4*loop_bw*damping_factor_squared + 2 + 2*damping_factor_squared
31  * b = - 8*loop_bw + 8*loop_bw*damping_factor_squared + 1 + 4*damping_factor_squared
32  * c = - 4*loop_bw
33  * \endcode
34  *
35  * Given rho, gamma is computed in closed form as
36  * \code gamma = ( 4*damping_factor_squared*rho ) / ( s_slope * (1+rho)^2 ) \endcode
37  */
38 lpf2 * ccsds_make_lpf2(const double loop_bw, const double damping_factor_squared, const double s_slope);
