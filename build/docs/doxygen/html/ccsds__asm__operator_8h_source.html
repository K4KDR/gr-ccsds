76  /*! \brief Checks if stream matches an ASM at given bit offset and
77  * return the number of bit errors found.
78  *
79  * \param symbols Array of unpacked bytes to check ASM against. Must
80  * contain at least d_ASM_SYM_LEN elements.
81  * \param offset_bits Number of bits to ignore in the first symbol.
82  * \return Hamming distance (number of bit positions that differ)
83  * between the symbols and the ASM.
84  */
85  unsigned int check_for_asm_hamming(const unsigned char* symbols, unsigned int offset_bits) {
86  #ifdef CCSDS_ASM_DEBUG
87  printf("++ check_for_asm in: ");
88  for(unsigned int i=0;i<std::ceil(d_ASM_LEN*8.0f/d_INFO_BITS_PER_SYMBOL);i++) {
89  printf("%02X ",symbols[i]);
90  }
91  printf(" with bit_offset=%u++\n",offset_bits);
92  #endif
93 
94  if(offset_bits >= d_INFO_BITS_PER_SYMBOL) {
95  fprintf(stderr,"WARNING ASM OPERATOR: invalid offset_bits %u.\n", offset_bits);
96  return false;
97  }
98 
99 
100  // Bitmasks to select a single bit position
101  unsigned char mask[8];
102  for(unsigned int i=0;i<8;i++) {
103  mask[i] = 0x1 << (7-i);
104  }
105 
106  // counter how many bits are still to be compared
107  unsigned int count = d_ASM_LEN*8u;
108 
109  // byte index in the stream where the current bit is.
110  unsigned int s_byte = 0;
111  // bit index in the current stream 0 means MSB
112  unsigned int s_bit = offset_bits;
113 
114  // byte index in the ASM that we are currently comparing
115  unsigned int a_byte = 0;
116  // bit index in the current ASM byte. 0 means MSB
117  unsigned int a_bit = 0;
118 
119  // Number of bit errors that we can still tolerate to
120  // consider sequence as ASM
121  unsigned int err_count = 0;
122 
123  // loop through all ASM bits
124  while(count > 0) {
125  const unsigned char bit_mask = (1 << (d_INFO_BITS_PER_SYMBOL-1)) >> s_bit;
126 
127  // current bit value in the stream moved to the
128  // LSB position
129  const unsigned char val_s_bit = (symbols[s_byte] & bit_mask) >> (d_INFO_BITS_PER_SYMBOL-1-s_bit);
130  // current bit value in the ASM moved to the LSB
131  // position
132  const unsigned char val_a_bit = ( d_ASM[a_byte] & mask[a_bit]) >> (7-a_bit);
133 
134  #ifdef CCSDS_ASM_DEBUG
135  printf(" bit %u %sequal, %u comparisons remaining\n",val_s_bit,((val_s_bit != val_a_bit) ? "not " : ""), count);
136  #endif
137 
138  // if bits are not equal, there is a ber
139  if(val_s_bit != val_a_bit) {
140  err_count++;
141  #ifdef CCSDS_ASM_DEBUG
142  printf(" bit error %u/%u\n",err_count, d_BER_THRESHOLD);
143  #endif
144  }
145 
146  s_bit++;
147  a_bit++;
148  count--;
149  if(s_bit >= d_INFO_BITS_PER_SYMBOL) {
150  s_bit = 0;
151  s_byte++;
152  }
153  if(a_bit >= 8) {
154  a_bit = 0;
155  a_byte++;
156  }
157  }
158 
159  #ifdef CCSDS_ASM_DEBUG
160  printf(" Check complete with %u bit errors\n", err_count);
161  printf("+++++++++++++++++++\n");
162  #endif
163 
164  return err_count;
165  }

183  /*! \brief Searches for ASM in the given stream.
184  *
185  * \param stream Array of bytes in which to search for the ASM.
186  * \param stream_len Length of the stream. The last \c d_ASM_SYM_LEN bytes of this
187  * stream are not searched (as the ASM would only fit in partially).
188  * \param *offset_bytes Pointer to an unsigned int where the offset bytes between
189  * start of stream and the found ASM should be stored, if a match is found.
190  * \param *offset_bits Pointer to an unsigned int where the offset bits between
191  * start of stream and the found ASM should be stored, if a match is found.
192  * \return \a true if ASM is found, \a false otherwise.
193  *
194  * The match with the minimum Hamming distance is searched. If this
195  * minimum Hamming distance is less or equal the \c d_BER_THRESHOLD a
196  * match is found, otherwise no match is found and \c d_offset_bytes
197  * and \c d_offset_bits are set to zero.
198  *
199  * In case there are two matches with the same Hamming distance the one
200  * with the lower byte offset or lower bit offset (in this priority) is
201  * chosen.
202  */
203  bool search_asm(const unsigned char* stream, const unsigned int stream_len, unsigned int *offset_bytes, unsigned int *offset_bits) {
204  *offset_bytes = 0;
205  *offset_bits = 0;
206  unsigned int min_hamming = d_BER_THRESHOLD+1; // To many bit errors
207  // check all possible byte offsets
208  for(unsigned int byte=0;byte<stream_len-d_ASM_SYM_LEN;byte++) {
209  // for a given byte offset, check all possible bit offsets
210  for(unsigned int bit=0;bit<d_INFO_BITS_PER_SYMBOL;bit++) {
211 
212  const unsigned int hamming_dist = check_for_asm_hamming(&stream[byte], bit);
213  // Does the stream at this position match the ASM?
214  if(hamming_dist < min_hamming && hamming_dist <= d_BER_THRESHOLD) {
215 
216  // Better match found, store position
217  *offset_bytes = byte;
218  *offset_bits = bit;
219 
220  min_hamming = hamming_dist;
221  }
222  }
223  }
224 
225  return (min_hamming <= d_BER_THRESHOLD);
226  }

229  /*! \brief Copy bitstream from a byte array with a bit offset into an
230  * aligned byte array.
231  *
232  * \param stream_in Array of bytes to copy. First bit to copy is
233  * located in the first element of this array. If \c offset_bit is
234  * zero this array must contain at least \c len elements, if
235  * \c offset_bit is greater thatn zero, this array must contain
236  * at least \c len+1 elements.
237  * \param stream_out Array in which the resulting bytes should be
238  * copied. Memory for at least \c len elements must be allocated.
239  * \param len Number of bytes to copy.
240  * \param offset_bits Number of bits to skip in the first byte, starting
241  * with the MSB. If offset_bit is zero this function copies the
242  * first \c len bytes from \c stream_in to \c stream_out by using
243  * \c memcpy. Otherwise the copy operation is done manually using
244  * bitwise operations.
245  *
246  *
247  * The first \c offset_bits starting with the MSB from the first
248  * element of \c stream_in are skipped, afterwards all bits from
249  * \c stream_in are copied to \c stream_out until \c len bytes have
250  * been copied to \c stream_out.
251  */
252  void copy_stream(unsigned char * stream_out, const unsigned char * stream_in, const unsigned int len, const unsigned int offset_bits) {
253 
254  if(offset_bits == 0u) {
255  // Input stream is aligned, use memcpy instead of doing
256  // everything by hand
257  memcpy(stream_out, stream_in, len*sizeof(unsigned char));
258  } else {
259  // Input stream has a bit offset, so we need to do the work
260 
261  // Go through all output bytes
262  for(unsigned int i=0;i<len;i++) {
263  // Shift the bits from the first byte to the right position
264  unsigned char tmp_l = stream_in[i] << offset_bits;
265  // Shift the bits from the second byte to the right position
266  unsigned char tmp_h = stream_in[i+1] >> (8-offset_bits);
267 
268  // Both bytes should now have all zeros on the positions
269  // where they don't carry information, we join them by
270  // bit-wise OR operations and write them out.
271  stream_out[i] = tmp_l | tmp_h;
272  }
273  }
274 
275  return;
276  }
